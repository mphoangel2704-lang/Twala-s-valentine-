    const box = document.getElementById("box");
    const noBtn = document.getElementById("noBtn");
    const yesBtn = document.getElementById("yesBtn");
    const result = document.getElementById("result");
    const countdownEl = document.getElementById("countdown");
    const heartsEl = document.getElementById("hearts");

    function moveNoButton() {
      const boxRect = box.getBoundingClientRect();
      const btnRect = noBtn.getBoundingClientRect();

      const padding = 10;
      const maxX = boxRect.width - btnRect.width - padding;
      const maxY = boxRect.height - btnRect.height - padding;

      const x = Math.random() * (maxX - padding) + padding;
      const y = Math.random() * (maxY - padding) + padding;

      noBtn.style.left = x + "px";
      noBtn.style.top = y + "px";
      noBtn.style.transform = "translate(0, 0)";
    }

    // Dodge interactions
    noBtn.addEventListener("mouseenter", moveNoButton);
    noBtn.addEventListener("touchstart", (e) => {
      e.preventDefault();
      moveNoButton();
    }, { passive: false });

    box.addEventListener("mousemove", (e) => {
      const btnRect = noBtn.getBoundingClientRect();
      const mx = e.clientX, my = e.clientY;
      const bx = btnRect.left + btnRect.width / 2;
      const by = btnRect.top + btnRect.height / 2;
      const dist = Math.hypot(mx - bx, my - by);
      if (dist < 90) moveNoButton();
    });

    yesBtn.addEventListener("click", () => {
      result.textContent = "YAYYYYY ðŸ˜­ðŸ’— See you on Valentineâ€™s Day!";
      noBtn.disabled = true;
      noBtn.style.opacity = 0.4;
      noBtn.style.cursor = "not-allowed";
    });

    noBtn.addEventListener("click", () => {

    // Countdown to Feb 14 (uses the viewerâ€™s local time)
    function updateCountdown(){
      const now = new Date();
      const year = now.getFullYear();
      let target = new Date(year, 1, 14, 0, 0, 0); // month is 0-based; 1 = Feb
      if (now > target) target = new Date(year + 1, 1, 14, 0, 0, 0);

      const ms = target - now;
      const totalSeconds = Math.max(0, Math.floor(ms / 1000));
      const days = Math.floor(totalSeconds / 86400);
      const hours = Math.floor((totalSeconds % 86400) / 3600);
      const mins = Math.floor((totalSeconds % 3600) / 60);

      countdownEl.textContent = `Countdown: ${days}d ${hours}h ${mins}m`;
    }
    setInterval(updateCountdown, 1000);
    updateCountdown();

    // Floating hearts generator
    function spawnHeart(){
      const h = document.createElement("div");
      h.className = "h";
      const left = Math.random() * 100; // vw
      const duration = 6 + Math.random() * 6; // seconds
      const size = 10 + Math.random() * 16; // px
      const opacity = 0.55 + Math.random() * 0.35;

      h.style.left = left + "vw";
      h.style.animationDuration = duration + "s";
      h.style.width = size + "px";
      h.style.height = size + "px";
      h.style.opacity = opacity;

      // match pseudo elements size
      h.style.setProperty("--s", size + "px");

      heartsEl.appendChild(h);

      // remove after animation
      setTimeout(() => h.remove(), duration * 1000);
    }

    // Spawn a few right away + keep spawning
    for (let i = 0; i < 10; i++) setTimeout(spawnHeart, i * 250);
    setInterval(spawnHeart, 450);

    });
